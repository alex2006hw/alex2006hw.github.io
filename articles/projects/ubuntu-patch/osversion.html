#!/bin/sh

site=$1
[ "$site" ] && [ "$site" != "hosting" ] && [ "$site" != "switch" ] && [ "$site" != "qa" ] && echo "usuage : $0 (hosting|switch|qa)" && exit 1
[ ! "$site" ]&& site='*'
grep Ubuntu ~/operations/inventory/${site}/*/issue | sed 's/.*inventory//' | sed 's/\/issue:Ubuntu//' | awk '{ printf"%-20s\t%s\n",$1,$2 }'
